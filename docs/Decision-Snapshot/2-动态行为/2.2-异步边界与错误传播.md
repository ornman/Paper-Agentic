```python
async def embed(self, texts: list[str]) -> list[list[float]]:
    """批量嵌入."""
    batch_size = settings.embedding_batch_size
    results = []

    for i in range(0, len(texts), batch_size):
        batch = texts[i:i + batch_size]
        response = await self._call_api(batch)
        results.extend(response["embeddings"])

    return results
```

```python
    """领域层统一错误基类."""
    code: str
```

## ￥问题￥ - 错误处理缺陷

### 1. ￥VLM 错误分类缺失￥

**事实**：`describer.py:84-88` 捕获所有异常

**问题**：
- 无法区分临时性错误（网络抖动）和永久性错误（文件损坏）
- 无法实现智能重试
- 用户体验差

**建议**：
```python
class VLMError(Exception):
    """VLM 调用错误."""
    def __init__(self, message: str, retryable: bool = False):
        self.retryable = retryable
        super().__init__(message)

async def _describe_one(index: int, chunk: Chunk):
    try:
        description = await vlm_client.describe_image(abs_path)
        return index, str(chunk.image_path), description
    except httpx.HTTPStatusError as e:
        if e.response.status_code in (429, 503, 504):
            # 临时性错误，可重试
            raise VLMError(f"VLM rate limited: {e}", retryable=True)
        else:
            # 永久性错误
            return index, str(chunk.image_path), "description unavailable"
```

### 2. ￥错误堆栈丢失￥

**事实**：`ingestion/service.py:113-121` 捕获所有异常

**问题**：
- `except Exception` 捕获所有异常
- 只保留 `message`，丢失堆栈
- 难以调试

**建议**：
```python
except Exception as exc:
    logger.exception("Unexpected error during ingestion")
    unexpected_error = IngestionError(
        code="ingestion_failed",
        message=f"导入链路执行失败: {exc}",
        detail={
            "document_id": current_record.document_id,
            "traceback": traceback.format_exc(),
        },
    )
    return self._mark_failed(current_record, unexpected_error)
```

### 3. ￥前端错误边界缺失￥

**事实**：所有组件都是 `try-catch` 在 store 层

**问题**：
- 组件层错误会导致整个应用崩溃
- 无错误边界组件
- 用户体验差

**建议**：
```vue
<template>
  <ErrorBoundary>
    <AppShell />
  </ErrorBoundary>
</template>
```

### 4. ￥SSE 错误恢复缺失￥

**事实**：`sse-client.ts` 无自动重连

**问题**：
- 网络抖动导致连接中断
- 用户需要手动刷新
- 已消耗的 token 浪费

**建议**：
```typescript
async function postAskInspirationStream(
  payload: AskInspirationRequestPayload,
  handlers: StreamHandlers,
  options: { maxRetries?: number } = {}
) {
  const { maxRetries = 3 } = options

  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      await _streamWithRetry(payload, handlers)
      return
    } catch (error) {
      if (attempt === maxRetries - 1) {
        handlers.onErrorEvent(error.message)
      } else {
        await new Promise(resolve => setTimeout(resolve, 1000 * (attempt + 1)))
      }
    }
  }
}
```

**1. 错误分类体系**

当前只有简单的 `DomainError`，未来可扩展为：

```python
class ErrorCategory(Enum):
    """错误分类."""
    TRANSIENT = "transient"  # 临时性错误（可重试）
    PERMANENT = "permanent"  # 永久性错误（不可重试）

class DomainError(Exception):
```

**2. 错误恢复策略**

```python
class RetryPolicy:
    """重试策略."""
    max_attempts: int = 3
    base_delay: float = 1.0
    max_delay: float = 60.0
    exponential_base: float = 2.0

async def retry_async(
    func: Callable,
    policy: RetryPolicy,
    error_categories: list[ErrorCategory] = [ErrorCategory.TRANSIENT],
):
    """🔮 带重试的异步调用."""
    for attempt in range(policy.max_attempts):
        try:
            return await func()
        except DomainError as e:
            if e.category not in error_categories:
                raise
            if attempt == policy.max_attempts - 1:
                raise
            delay = min(
                policy.base_delay * (policy.exponential_base ** attempt),
                policy.max_delay
            )
            await asyncio.sleep(delay)
```

## 错误传播路径图

```
[底层错误]
    ↓
[领域错误] (DomainError)
    ├─ ConfigError
    ├─ IngestionError
    ├─ IndexingError
    ├─ RetrievalError
    └─ QAError
    ↓
[HTTP 错误] (FastAPI HTTPException)
    ├─ 400: 请求错误
    ├─ 401: 认证失败
    ├─ 429: 限流
    ├─ 500: 服务器错误
    └─ 503: 服务不可用
    ↓
[前端错误] (ApiClientError)
    ├─ statusCode: HTTP 状态码
    └─ message: 错误消息
    ↓
[UI 错误] (用户可见)
    └─ 友好的错误提示
```
