```python
@model_validator(mode="after")
def validate_pinned_model_contract(self):
    """校验模型契约，防止向量维度和模型漂移污染索引。"""
    if self.embedding_model != PINNED_EMBEDDING_MODEL:
        raise ValueError(f"EMBEDDING_MODEL 必须固定为 {PINNED_EMBEDDING_MODEL}")
    if self.embedding_dimensions != PINNED_EMBEDDING_DIMENSIONS:
        raise ValueError(f"EMBEDDING_DIMENSIONS 必须固定为 {PINNED_EMBEDDING_DIMENSIONS}")
    if self.rerank_model != PINNED_RERANK_MODEL:
        raise ValueError(f"RERANK_MODEL 必须固定为 {PINNED_RERANK_MODEL}")
    return self
```

## 四、代码组织模式

### 4.1 模块化结构

**【代码事实】领域驱动拆分** (`modules/`):

| 领域模块 | 职责 | 对外接口 | 内部结构 |
|---------|------|----------|----------|

| `library` | 文档库管理 | `LibraryService.import_pdf()` | `models.py`, `repository.py`, `service.py` |

- ❌ 部分模块职责模糊（如`library`vs`ingestion`）

```python
class Container(containers.DeclarativeContainer):
    config = providers.Configuration()

    qdrant_store = providers.Singleton(
        QdrantStore,
        path=config.qdrant_path,
    )

    embedding_client = providers.Singleton(
        SiliconFlowEmbeddingClient,
    )

    retrieval_service = providers.Factory(
        RetrievalService,
        qdrant_store=qdrant_store,
        embedding_client=embedding_client,
    )

# 使用
@app.post("/query/retrieve")
async def retrieve(
    service: RetrievalService = Depends(lambda: container.retrieval_service),
```
